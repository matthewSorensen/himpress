mergesort [] = []
mergesort [x] = [x]
mergesort x  = let (l,r) = foldl (\(x,y) z -> (z:y,x)) ([],[]) x
               in merge (mergesort l) (mergesort r)
  where merge [] x = x
        merge x [] = x
        merge (x:xs) (y:ys)
          | x <= y = x : merge xs (y:ys)
          | otherwise = y : merge (x:xs) ys
